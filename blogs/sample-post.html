Python Example


def fibonacci(n):
    """Return the n-th Fibonacci number."""
    if n <= 1:
        return n
    return fibonacci(n-1) + fibonacci(n-2)

print(fibonacci(10))  # Prints the 10th Fibonacci number

x86 Assembly Example


section .data
    msg db "Hello, Assembly!", 0

section .text
    global _start

_start:
    ; write() system call
    mov eax, 4      ; sys_write
    mov ebx, 1      ; file descriptor: stdout
    mov ecx, msg    ; message to write
    mov edx, 17     ; message length
    int 0x80
    
    ; exit() system call
    mov eax, 1      ; sys_exit
    xor ebx, ebx    ; status = 0
    int 0x80
